product_id BIGINT UNSIGNED NOT NULL,\n    image VARCHAR(255) NOT NULL,\n    is_main BOOLEAN DEFAULT FALSE,\n    order INT DEFAULT 0,\n    created_at TIMESTAMP,\n    updated_at TIMESTAMP,\n    FOREIGN
